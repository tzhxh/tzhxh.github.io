对开发中遇到的问题进行一些总结，记录关于封装的一些要点

将要封装的内容写到一个 .js 文件中,在需要用到的HTML页面中用 script标签引入就可使用，

首先创建一个静态类，可以不需要实例化直接使用 

var common = function () {};

然后为这个类添加属性 

common.xxx = undefined/null/boolean/number/string   //五种基本数据类型
common.xxx = {}                                     //object 引用数据类型

此处封装了ajax ， 使用的是原生ajax。完整代码如下； 

var common = function () {};

common.service = { post: function (url, data, callback) { let request = new XMLHttpRequest(); request.open("post", url, true); request.setRequestHeader("Content-type", "application/json;charset=utf-8"); request.send(JSON.stringify(data)); request.onreadystatechange = function () { if (request.readyState == 4) { if (request.status == 200) { callback(request.responseText); } } }; }, get: function (url, data, callback) { let request = new XMLHttpRequest(); request.open("get", url, true); request.setRequestHeader("Content-type", "application/json;charset=utf-8"); request.send(JSON.stringify(data)); request.onreadystatechange = function () { if (request.readyState == 4) { if (request.status == 200) { callback(request.responseText); } } }; }, };

一共三个参数 url为路径 ， data为要发送的json数据 ，callback 为回调函数 以返回的数据作为实参 在调用时可以按如下方法使用

common.service.post(url,data,function (data) { //todo url填入路径类型为string当前为登录路径 ,data 为一个对象{key:value,key:value}

        })
