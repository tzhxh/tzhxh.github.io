jsp+servlet+javabean实现对数据库的操作

发表于 2019-04-20

用jsp+servlet+Javabean实现登录跳转并对数据库进行操作

其中jsp用以和用户交互，servlet起到控制器的作用，Javabean对后台进行操作。

工作流程


1.要对数据库进行操作首先要找到对应的数据库的JDBC，我使用的数据库为MySQL
对应的JDBC文件在MySQL文件夹下的Connector J 8.0文件夹下名为mysql-connector-java-8.0.15
完整的路径为‪C:\Program Files (x86)\MySQL\Connector J 8.0\mysql-connector-java-8.0.15.jar
将其导入到编写Javabean文件的编译器中，并将其复制到Tomcat的lib文件夹下

2.在Tomcat的webapps文件夹下新建一个工程文件夹，可随意命名，我命名为login，将jsp文件存放在该文
件夹下

3.在login文件夹下新建一个文件夹 命名为WEB-INF，该文件夹用来存放配置文件 web.xml

4.在WEB-INF文件夹下再新建一个文件夹 命名为classes 用来存放servlet文件和Javabean文件
若文件有包的话，连同包的文件夹一起放在classes下 若无包就直接存放 .class文件

简单示例


‪先打开一个登陆页面，如有账号则输入用户名和密码登录，若无账号，则点击注册进入到注册页面，输入用户
名和密码，并将用户名和密码存入数据库中，显示注册成功并返回登录页面，再输入用户名和密码，查询数据
库中是否有用户名和密码并验证是否正确，若用户存在且密码正确则跳转到工作页面，此处工作页面为问卷调
差，将收集登录者填写的问卷信息存入数据库中，跳转到提交成功页面。
包括以下文件 Javabean文件(User.java),(UserDao.java)和(DataBaseUtil.java)
servlet文件 (loginservlet.java),(regservlet.java)和(getanwservlet.java)
jsp 文件 (login.jsp)(login.css),(register.jsp)(register.css),(page.jsp)和(end.jsp)

import java.sql.Connection;
import java.sql.*;

public class UserDao {
    //判用户是否存在断
public boolean userExist(String username)
{
    Connection conn = DataBaseUtil.getConn_u();
    //根据用户名查询信息
    String sql = "select * from user_info where username = ?";

    try {
        PreparedStatement preparedStatement = conn.prepareStatement(sql);
        preparedStatement.setString(1, username);
        ResultSet resultSet = preparedStatement.executeQuery();
        while (!resultSet.next())
        {
            return false;
        }
        resultSet.close();
        preparedStatement.close();
    }catch(SQLException e) {
        e.printStackTrace();
    }finally {
        DataBaseUtil.closeConn(conn);
    }
    return true;
}
//注册成功 则提交信息并存入数据库
public void saveUser(User user)
{
    //获取数据库连接
    Connection conn = DataBaseUtil.getConn_u();
    String sql = "insert into user_info(username,password) values(?,?)";
    try {
        PreparedStatement ps = conn.prepareStatement(sql);
        ps.setString(1, user.getUsername());
        ps.setString(2, user.getPassword());
        ps.executeUpdate();
        ps.close();
    }catch(SQLException e) {
        e.printStackTrace();
    }
}

//登录
public User login(String username,String password) {
    User user = null;
    Connection conn = DataBaseUtil.getConn_u();
    String sql = "select * from user_info where username=? and password=?";
    try {
        PreparedStatement ps = conn.prepareStatement(sql);
        ps.setString(1, username);
        ps.setString(2, password);
        ResultSet rs = ps.executeQuery();

        while(rs.next()) {
            user = new User();
            user.setId(rs.getInt("id"));
            user.setUsername(rs.getString("username"));
            user.setPassword(rs.getString("password"));
        }
        rs.close();
        ps.close();
    }catch(SQLException e)
    {
        e.printStackTrace();
    }finally {
        DataBaseUtil.closeConn(conn);
    }
    return user;
}

public void saveAnswer(User user) {
    Connection conn = DataBaseUtil.getConn_q();

import java.io.*;
import javax.servlet.ServletException;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

public class regservlet extends HttpServlet{

protected void doPost(HttpServletRequest req, HttpServletResponse resp)throws ServletException, IOException 
{
    resp.setContentType("text/html;charset=utf-8");
    req.setCharacterEncoding("utf-8");
    //获取注册信息
    String username = req.getParameter("username");
    String password = req.getParameter("password");
    String password_r = req.getParameter("password_r");
    System.out.println(password +"," +password_r);
    if(!username.equals("")) {
        if(password.equals(password_r))
        {
            UserDao userdao = new UserDao();
            if(!userdao.userExist(username)) {
                User user = new User();
                user.setUsername(username);
                user.setPassword(password);
                userdao.saveUser(user);
                req.setAttribute("info","注册成功！<br>");
                req.getRequestDispatcher("login.jsp").forward(req, resp);
            }else {
                req.setAttribute("info", "此用户已存在！ <br>");
                req.getRequestDispatcher("register.jsp").forward(req, resp);
            }
        }else {
            req.setAttribute("info", "密码不一致! <br>");
            req.getRequestDispatcher("register.jsp").forward(req, resp);
        }
    }else {
        req.setAttribute("info", "用户名不能为空! <br>");
        req.getRequestDispatcher("register.jsp").forward(req, resp);
    }


}
protected void doGet(HttpServletRequest req,HttpServletResponse resp) throws ServletException, IOException
{
    doPost(req ,resp);
}
}

5.loginservlet.java

servlet文件 实现登录功能
代码如下：

点击显示/隐藏代码


import java.io.*;
import javax.servlet.ServletException;
import javax.servlet.http.HttpServlet;
import javax.servlet.http.HttpServletRequest;
import javax.servlet.http.HttpServletResponse;

public class loginservlet extends HttpServlet{
    protected void doPost(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        resp.setContentType("text/html;charset=utf-8");
        req.setCharacterEncoding("utf-8");

        String username = req.getParameter("username");
        String password = req.getParameter("password");
        if(!username.equals("")) {
            UserDao userDao = new UserDao();
            //根据密码查询用户
            User user  = userDao.login(username, password);
            //判断user是否为空
            if (user != null) {
                req.setAttribute("user", username);
                //转发到result.jsp页面
                req.getRequestDispatcher("page.jsp").forward(req, resp);
                /**
                 response.sendRedirect(url)跳转到指定的URL地址，产生一个新的request，所以要传递参数只有在url后加参
                 数，如：
                 url?id=1.
                 request.getRequestDispatcher(url).forward(request,response)是直接将请求转发到指定URL，所以该请求
                 能够直接获得上一个请求的数据，也就是说采用请求转发，request对象始终存在，不会重新创建。而
                 sendRedirect()会新建request对象，所以上一个request中的数据会丢失。
                 */
            }else {
                //登录失败
                req.setAttribute("info","用户名或密码错误！");
                req.getRequestDispatcher("login.jsp").forward(req, resp);
            }
        }else {
            req.setAttribute("info","用户名不能为空！");
            req.getRequestDispatcher("login.jsp").forward(req, resp);
        }


}
    protected void doGet(HttpServletRequest req, HttpServletResponse resp) throws ServletException, IOException {
        doPost(req, resp);
    }
}

在Tomcat的bin文件夹下找到startup.bat文件双击即可开启服务
开启服务后在浏览器中输入路径http://localhost:8080/login/login.jsp
其中localhost是本地的ip，8080为Tomcat的默认端口，webapps下的工程文件夹的名字(根据自己建立的文件夹来修改),login.jsp是要打开的页面(根据自己的名称修改)
结束时双击bin文件夹下的shutdown.bat来关闭服务
